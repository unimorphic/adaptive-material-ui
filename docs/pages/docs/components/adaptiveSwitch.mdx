# Adaptive Switch

## Adaptiveness

- iOS: native styling

```tsx
import AdaptiveModeDemo from "@/shared/adaptiveModeDemo";
import Stack from "@mui/material/Stack";
import { AdaptiveSwitch } from "adaptive-material-ui/components/switch/adaptiveSwitch";

export default function () {
  return (
    <AdaptiveModeDemo>
      <AdaptiveSwitch defaultChecked />
    </AdaptiveModeDemo>
  );
}
```

## iOS Variants

```tsx
import AdaptiveModeDemo from "@/shared/adaptiveModeDemo";
import { DemoList, DemoListItem } from "@/shared/demoList";
import FormControlLabel from "@mui/material/FormControlLabel";
import { AdaptiveSwitch } from "adaptive-material-ui/components/switch/adaptiveSwitch";

export default function () {
  return (
    <AdaptiveModeDemo adaptiveMode="ios">
      <DemoList>
        <DemoListItem title="Secondary Color">
          <AdaptiveSwitch color="secondary" defaultChecked />
        </DemoListItem>
        <DemoListItem title="Size">
          <FormControlLabel
            control={<AdaptiveSwitch defaultChecked size="small" />}
            label="Small"
          />
          <FormControlLabel
            control={<AdaptiveSwitch defaultChecked size="medium" />}
            label="Medium"
          />
        </DemoListItem>
        <DemoListItem title="Disabled">
          <AdaptiveSwitch defaultChecked disabled />
        </DemoListItem>
      </DemoList>
    </AdaptiveModeDemo>
  );
}
```

## API

[MUI Switch API](https://mui.com/material-ui/api/switch/) with the following changes:

- `disableRipple` defaults to `true` on iOS
