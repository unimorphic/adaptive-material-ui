## Customization

The iOS mode can be customized using the iOS class or theme as shown below. The other modes use the standard [Material UI Switch](https://mui.com/material-ui/react-switch/#customization).

### iOS Custom Class

```tsx
import BasicTheme from "@/components/basicTheme";
import { styled } from "@mui/material/styles";
import { switchClasses } from "@mui/material/Switch";
import { AdaptiveMode } from "adaptive-material-ui/adaptiveMode/adaptiveMode";
import AdaptiveSwitch from "adaptive-material-ui/components/adaptiveSwitch";

const AdaptiveSwitchStyled = styled(AdaptiveSwitch)({
  "&.ios": {
    [`& .${switchClasses.track}`]: {
      backgroundColor: "rebeccapurple",
    },

    [`& .${switchClasses.checked} + .${switchClasses.track}`]: {
      backgroundColor: "purple",
    },
  },
});

export default function Customization_Classes() {
  return (
    <BasicTheme>
      <AdaptiveSwitchStyled
        adaptiveMode={AdaptiveMode.ios}
        classes={{ ios: "ios" }}
        defaultChecked
      />
    </BasicTheme>
  );
}
```

### iOS Custom Theme

```tsx
import BasicTheme from "@/components/basicTheme";
import { createTheme, ThemeProvider } from "@mui/material/styles";
import { switchClasses } from "@mui/material/Switch";
import { AdaptiveMode } from "adaptive-material-ui/adaptiveMode/adaptiveMode";
import AdaptiveSwitch from "adaptive-material-ui/components/adaptiveSwitch";
import type {} from "adaptive-material-ui/themeAugmentation";

function customizeTheme(theme: Theme) {
  return createTheme({
    ...theme,
    components: {
      ...theme.components,
      AdaptiveSwitch: {
        styleOverrides: {
          ios: {
            [`& .${switchClasses.track}`]: {
              backgroundColor: "lightblue",
            },

            [`& .${switchClasses.checked} + .${switchClasses.track}`]: {
              backgroundColor: "blue",
            },
          },
        },
      },
    },
  });
}

export default function Customization_Theme() {
  return (
    <BasicTheme>
      <ThemeProvider theme={customizeTheme}>
        <AdaptiveSwitch adaptiveMode={AdaptiveMode.ios} defaultChecked />
      </ThemeProvider>
    </BasicTheme>
  );
}
```
