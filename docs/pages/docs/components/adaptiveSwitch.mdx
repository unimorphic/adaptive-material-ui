# Adaptive Switch

## Adaptiveness

- iOS: native styling

```tsx
import AdaptiveModeDemo from "@/shared/adaptiveModeDemo";
import Stack from "@mui/material/Stack";
import { AdaptiveSwitch } from "adaptive-material-ui/components/switch/adaptiveSwitch";
import { useState } from "react";

export default function () {
  const [adaptiveMode, setAdaptiveMode] = useState("");

  return (
    <AdaptiveModeDemo adaptiveMode={adaptiveMode} onChange={setAdaptiveMode}>
      <AdaptiveSwitch adaptiveMode={adaptiveMode} defaultChecked />
    </AdaptiveModeDemo>
  );
}
```

## iOS Variants

```tsx
import FormControlLabel from "@mui/material/FormControlLabel";
import Grid from "@mui/material/Grid";
import { AdaptiveSwitch } from "adaptive-material-ui/components/switch/adaptiveSwitch";

export default function () {
  return (
    <Grid container direction="row" spacing={2}>
      <Grid>
        <FormControlLabel
          control={
            <AdaptiveSwitch
              adaptiveMode="ios"
              color="secondary"
              defaultChecked
            />
          }
          label="Secondary Color"
        />
      </Grid>
      <Grid>
        <FormControlLabel
          control={
            <AdaptiveSwitch adaptiveMode="ios" defaultChecked disabled />
          }
          label="Disabled"
        />
      </Grid>
      <Grid>
        <FormControlLabel
          control={
            <AdaptiveSwitch adaptiveMode="ios" defaultChecked size="small" />
          }
          label="Small"
        />
      </Grid>
    </Grid>
  );
}
```
