## iOS Variants

```tsx
import Grid from "@mui/material/Grid";
import { AdaptiveMode } from "adaptive-material-ui/adaptiveMode";
import AdaptiveSlider from "adaptive-material-ui/components/slider/adaptiveSlider";

export default function Basic() {
  return (
    <Grid container spacing={2}>
      <Grid size={3}>Secondary Color</Grid>
      <Grid size={9}>
        <AdaptiveSlider
          adaptiveMode={AdaptiveMode.ios}
          color="secondary"
          defaultValue={30}
        />
      </Grid>
      <Grid size={3}>Disabled</Grid>
      <Grid size={9}>
        <AdaptiveSlider
          adaptiveMode={AdaptiveMode.ios}
          defaultValue={30}
          disabled
        />
      </Grid>
      <Grid size={3}>Small</Grid>
      <Grid size={9}>
        <AdaptiveSlider
          adaptiveMode={AdaptiveMode.ios}
          defaultValue={30}
          size="small"
        />
      </Grid>
      <Grid size={3}>Discrete</Grid>
      <Grid size={9}>
        <AdaptiveSlider
          adaptiveMode={AdaptiveMode.ios}
          defaultValue={30}
          marks
          shiftStep={30}
          step={10}
        />
      </Grid>
      <Grid size={3}>Vertical</Grid>
      <Grid size={9} sx={{ height: 100 }}>
        <AdaptiveSlider
          adaptiveMode={AdaptiveMode.ios}
          defaultValue={40}
          marks
          orientation="vertical"
          shiftStep={40}
          step={20}
        />
      </Grid>
    </Grid>
  );
}
```
