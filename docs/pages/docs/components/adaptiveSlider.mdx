# Adaptive Slider

## Adaptiveness

- iOS: native styling

```tsx
import AdaptiveModeDemo from "@/shared/adaptiveModeDemo";
import Stack from "@mui/material/Stack";
import { AdaptiveSlider } from "adaptive-material-ui/components/slider/adaptiveSlider";
import { useState } from "react";

export default function () {
  const [adaptiveMode, setAdaptiveMode] = useState("");

  return (
    <AdaptiveModeDemo
      alignItems="stretch"
      adaptiveMode={adaptiveMode}
      onChange={setAdaptiveMode}
    >
      <AdaptiveSlider adaptiveMode={adaptiveMode} defaultValue={30} />
    </AdaptiveModeDemo>
  );
}
```

## iOS Variants

```tsx
import Grid from "@mui/material/Grid";
import { AdaptiveSlider } from "adaptive-material-ui/components/slider/adaptiveSlider";

export default function () {
  return (
    <Grid container spacing={2}>
      <Grid size={3}>Secondary Color</Grid>
      <Grid size={9}>
        <AdaptiveSlider
          adaptiveMode="ios"
          color="secondary"
          defaultValue={30}
        />
      </Grid>
      <Grid size={3}>Disabled</Grid>
      <Grid size={9}>
        <AdaptiveSlider adaptiveMode="ios" defaultValue={30} disabled />
      </Grid>
      <Grid size={3}>Small</Grid>
      <Grid size={9}>
        <AdaptiveSlider adaptiveMode="ios" defaultValue={30} size="small" />
      </Grid>
      <Grid size={3}>Discrete</Grid>
      <Grid size={9}>
        <AdaptiveSlider
          adaptiveMode="ios"
          defaultValue={30}
          marks
          shiftStep={30}
          step={10}
        />
      </Grid>
      <Grid size={3}>Vertical</Grid>
      <Grid size={9} sx={{ height: 100 }}>
        <AdaptiveSlider
          adaptiveMode="ios"
          defaultValue={40}
          marks
          orientation="vertical"
          shiftStep={40}
          step={20}
        />
      </Grid>
    </Grid>
  );
}
```
