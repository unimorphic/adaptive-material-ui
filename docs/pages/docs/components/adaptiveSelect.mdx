# Adaptive Select

## Adaptiveness

- Desktop: doesn't block interaction with other page elements when open
- Android & iOS: uses the native device select
- iOS: native styling

```tsx
import AdaptiveModeSelector from "@/shared/AdaptiveModeSelector";
import Stack from "@mui/material/Stack";
import AdaptiveSelect, {
  AdaptiveSelectItem,
} from "adaptive-material-ui/components/select/adaptiveSelect";
import { useState } from "react";

export default function Basic() {
  const [adaptiveMode, setAdaptiveMode] = useState("");

  return (
    <Stack alignItems="flex-start" spacing={3}>
      <AdaptiveModeSelector
        adaptiveMode={adaptiveMode}
        onChange={setAdaptiveMode}
      />
      
      <AdaptiveSelect adaptiveMode={adaptiveMode} defaultValue={1}>
        <AdaptiveSelectItem value={1}>One</AdaptiveSelectItem>
        <AdaptiveSelectItem value={2}>Two</AdaptiveSelectItem>
      </AdaptiveSelect>
    </Stack>
  );
}
```
