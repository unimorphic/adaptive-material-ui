# Styles

## iOS Colors

The components use the theme palette colors for iOS components by default,
to use iOS native colors you can use `iosColorsTheme`

```tsx
import { createTheme, ThemeProvider } from "@mui/material/styles";
import { AdaptiveMode } from "adaptive-material-ui/adaptiveMode";
import AdaptiveSwitch from "adaptive-material-ui/components/switch/adaptiveSwitch";
import iosColorsTheme from "adaptive-material-ui/themes/iosColorsTheme";

const yourTheme = createTheme({ colorSchemes: { dark: true } });

export default function IOSColors() {
  return (
    <ThemeProvider theme={createTheme(iosColorsTheme, yourTheme)}>
      <AdaptiveSwitch adaptiveMode={AdaptiveMode.ios} defaultChecked />
    </ThemeProvider>
  );
}
```

## Component Styles

Components with device specific styles can be customized using the related classes prop or theme style override as shown below.
Other styles can be customized using the standard Material UI methods, for example:
[Themed Components](https://mui.com/material-ui/customization/theme-components/),
[Switch Customization](https://mui.com/material-ui/react-switch/#customization)

### iOS Custom Class

```tsx
import { styled } from "@mui/material/styles";
import { switchClasses } from "@mui/material/Switch";
import { AdaptiveMode } from "adaptive-material-ui/adaptiveMode";
import AdaptiveSwitch from "adaptive-material-ui/components/switch/adaptiveSwitch";

const AdaptiveSwitchStyled = styled(AdaptiveSwitch)({
  "&.ios": {
    [`& .${switchClasses.switchBase}:before, & .${switchClasses.track}`]: {
      backgroundColor: "rebeccapurple",
    },
    [`& .${switchClasses.switchBase}.${switchClasses.checked}`]: {
      [`&:before, & + .${switchClasses.track}`]: {
        backgroundColor: "purple",
      },
    },
  },
});

export default function Customization_Classes() {
  return (
    <AdaptiveSwitchStyled
      adaptiveMode={AdaptiveMode.ios}
      classes={{ ios: "ios" }}
      defaultChecked
    />
  );
}
```

### iOS Custom Theme

```tsx
import { createTheme, ThemeProvider } from "@mui/material/styles";
import { switchClasses } from "@mui/material/Switch";
import { AdaptiveMode } from "adaptive-material-ui/adaptiveMode";
import AdaptiveSwitch from "adaptive-material-ui/components/switch/adaptiveSwitch";
import type {} from "adaptive-material-ui/themes/themeAugmentation";

const theme = createTheme({
  colorSchemes: { dark: true },
  components: {
    AdaptiveSwitch: {
      styleOverrides: {
        ios: {
          [`& .${switchClasses.switchBase}:before, & .${switchClasses.track}`]:
            {
              backgroundColor: "lightblue",
            },
          [`& .${switchClasses.switchBase}.${switchClasses.checked}`]: {
            [`&:before, & + .${switchClasses.track}`]: {
              backgroundColor: "blue",
            },
          },
        },
      },
    },
  },
});

export default function Customization_Theme() {
  return (
    <ThemeProvider theme={theme}>
      <AdaptiveSwitch adaptiveMode={AdaptiveMode.ios} defaultChecked />
    </ThemeProvider>
  );
}
```
