# useAdaptiveMode

The `useAdaptiveMode` hook can be used to get the current adaptive mode based on the current context & device.
Examples of different usages of this hook are shown below.

## Component Prop

This example shows how to change a component prop based on the current adaptive mode,
the button will be `round` only on iOS devices.

```tsx pure
import { useAdaptiveMode } from "adaptive-material-ui/adaptiveMode";
import { AdaptiveButton } from "adaptive-material-ui/components/button/adaptiveButton";

export default function () {
  const adaptiveMode = useAdaptiveMode();

  return <AdaptiveButton round={adaptiveMode === "ios"}>Button</AdaptiveButton>;
}
```

## Theme

This example shows how to change the theme based on the current adaptive mode,
all buttons will default to `round` only on iOS devices.

```tsx pure
import { createTheme, ThemeProvider } from "@mui/material/styles";
import {
  AdaptiveMode,
  useAdaptiveMode,
} from "adaptive-material-ui/adaptiveMode";
import { AdaptiveButton } from "adaptive-material-ui/components/button/adaptiveButton";
import { useState } from "react";

function createMyTheme(adaptiveMode: AdaptiveMode) {
  return createTheme({
    components: {
      AdaptiveButton: {
        defaultProps: { round: adaptiveMode === "ios" },
      },
    },
  });
}

export default function () {
  const adaptiveMode = useAdaptiveMode();
  const [theme] = useState(() => createMyTheme(adaptiveMode));

  return (
    <ThemeProvider theme={theme}>
      <AdaptiveButton>Button</AdaptiveButton>
    </ThemeProvider>
  );
}
```
